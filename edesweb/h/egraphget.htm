eGraphGet
SINTAXIS
eGraphGet( ... )
DESCRIPCION
Esta función devuelve:
- Si no lo puede ejecutar = false
- Si lo puede ejecutar, un array con los siguientes valores:
   - 0.- La imagen
   - 1.- Ancho
   - 2.- Alto
   - 3.- Nombre completo del fichero temporal en el servidor
PARAMETROS
Tipo de gráfica      - R: Solo muestra por cada ROW una columna
    - C: Muestra la sumas por las COL indicadas
    - P: Muestra la sumas por las COL indicadas
    - G: Muestra todos los valores, en Regilla
 Un array con todos los valores a representar  
 Un array con los datos de los campos     - Lable
    - Nº de decimales
    - Un "+" ( si es cadena vacia no se tiene encuenta el dato)
 TITULO GRAFICA  
 TITULO COL  
 TITULO ROW  
 TITULO LEYENDA  
  Opcional un array para cambiar la comfiguración  
 
EJEMPLO

[PHPEnd] l

$gra = eGraphGet(     // Muestra todos los valores
'G',
array(
array(1000,500,700,20), // Se puede poner una sola fila con la suma
array(1500,200,900,50),
array( 500, 50,200,75),
array( 500, 50,200,10)
),
array(
array('Label_11',2,'+'),
array('Label_22',2,'+'),
array('Label_33',2,'+'),
array('Label_44',2,'+')
),
'TITULO GRAFICA',
'TITULO COL',
'TITULO ROW',
'TITULO LEYENDA'
);
if( $gra ) echo $gra[0];

$gra = eGraphGet(     // Muestra la sumas por las COL indicadas
'P',
array(
array(1000,500,700,20), // Se puede poner una sola fila con la suma
array(1500,200,900,50),
array( 500, 50,200,75),
array( 500, 50,200,10)
),
array(
array('Label_11',2,'+'),
array('Label_22',2,'+'),
array('Label_33',2,'+'),
array('Label_44',2,'+')
),
'TITULO GRAFICA',
'TITULO COL',
'',
'TITULO LEYENDA'
);
if( $gra ) echo $gra[0];


$gra = eGraphGet(  // Muestra la sumas por las COL indicadas
'C',
array(
array(1000,500,700), // Se puede poner una sola fila con la suma
array(1500,350,900),
array( 500, 50,200)
),
array(
array('Label_11',2,'+'),  // El "+" indica columna a representar
array('Label_22',2,'+'),
array('Label_33',2,'+')
),
'TITULO GRAFICA',
'TITULO COL',
'TITULO ROW'
);
if( $gra ) echo $gra[0];


$gra = eGraphGet(  // Solo muestra por cada ROW una columna
'R',
array(   // Valor
array('Label_1',1000),
array('Label_2',1500),
array('Label_3', 500)
),
array(
array('',2,''),
array('',2,'+')  // Columna a representar
),
'TITULO GRAFICA',
'TITULO COL',
'TITULO ROW'
);
if( $gra ) echo $gra[0];
