ePPF
SINTAXIS
ePPF( NombreFields, Value, ConChange, [OkReadyState=true] )
DESCRIPCION
Es igual que la función "ePF()" pero asigna los valores en la ventana Padre ( ePPF() = ParentPutFields )
PARAMETROS
EJEMPLO

[JSSelRow] *
ePPF( 'cd_prop',       eGF('cd_prop')  );
ePPF( '_pro_dni',       eGF('dni')       );
ePPF( '_pro_nombre',    eGF('nombre')    );
ePPF( '_pro_apellidos', eGF('apellidos') );   // Coge del listado "eGF('apellidos')" el valor de los apellidos y lo    graba en la ventana padre en el campo "_pro_apellidos"
