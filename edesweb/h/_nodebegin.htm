.nodeBegin
SINTAXIS
.nodeBegin( [objParent] )
DESCRIPCION
 
PARAMETROS
 
EJEMPLO
S("<span style='display:inline-block;border:1px solid green'>green</span>").nodeBegin();                   // inserta el span al principio del body
S("<span style='display:inline-block;border:1px solid green'>green</span>").nodeBegin(S("#prueba").obj);   // inserta el span al principio del objeto S("#prueba").obj
S("<span style='display:inline-block;border:1px solid green'>green</span>").nodeBegim("#prueba");          // el objeto se puede definir como cadena, igual que la función S()
