eBar
SINTAXIS
eBar( $color, $value )
DESCRIPCION
Encapsula un dato dentro de un botón
PARAMETROS
 $color b: blue
c: cyan
d: dark
g: green
l: lightgray
o: orange
p: purple
r: red
s: silver
w: wait
 $value Cualquier dato
EJEMPLO
[Format]
     if( $_vF["cargo"]=="SOCIA" ){
          eButton("d", $_vF["cargo"]);           // dark
     }else if( $_vF["cargo"]=="SOCIO" ){
           eButton("b", $_vF["cargo"]);         // blue
     }else{
           eButton("r", $_vF["cargo"]);         // red
     }
}
